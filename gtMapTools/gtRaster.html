140        '''
141        Get the data from the tiff file.
142        :param datafilename: File name of the tif you what to import to the raster.
143        :param datafilenames: List containing the names of all the file names you what to operate for the raster (overrides filename).
144        :param fileext: File extension text in case you use a different extension from .tif or .tiff.
145        :param filepath: Path where your data file or files are in. If no filename or filenames are provided, then all the tif files in this directory are used.
146        :param operation: Operation betwen tif (when multiple files are provided). ['acum'|'avr'|'median']

149        if datafilenames == None:
150            datafilenames = self.dtf_names

153            datafilename = self.dtf_name
154        
155        if filepath == None:
156            filepath = self.dtf_path
157        
158        if operation == None:
159            operation = self.opr
160        
161        self.dtf_names = datafilenames
162        self.dtf_name = datafilename
163        self.dtf_path = fullpathname(filepath)
164
165        if self.dtf_names == [] and self.dtf_name == None:
166            for file in os.listdir(self.dtf_path):
167                if fileext == None:
168                    if file.endswith('.tif') or file.endswith('.tiff'):
169                        self.dtf_names.append(file)
170                else:
171                    if file.endswith(fileext):
172                        self.dtf_names.append(file)
173        elif self.dtf_names == [] and self.dtf_name != None:
174            self.dtf_names.append(self.dtf_name)
175
176        if len(self.dtf_names) > 1: 
177
178            try:
179                d, latitude, longitude = tif_to_array(self.dtf_names[0], self.dtf_path)
180                self.longitudearray = longitude
181                self.latitudearray = latitude
182            except:
183                raise Exception("Couldn't find a valid tif file")
184
185            temp_dataarray = []
186
187            for file in self.dtf_names:
188                data, lat, lon = tif_to_array(file, self.dtf_path)
189                temp_dataarray.append(data)
190
191                # if self.longitudearray != lon or self.latitudearray != lat:
192                if not [np.array_equal(self.longitudearray, lon, equal_nan=True) or np.array_equal(self.latitudearray, lat, equal_nan=True)]:
193                    raise Exception("The coordinates on diferent tif files are different. This may cause errors.")
194            
195            # print(temp_dataarray)
196
197            if self.opr == 'acum' or self.opr == 'sum':
198                self.dataarray = sum(temp_dataarray)
199                # print(self.dataarray)
200            elif self.opr == 'mean' or self.opr == 'avr' or self.opr == 'average':
201                temp_dataarray = np.stack(temp_dataarray)
202                self.dataarray = np.mean(temp_dataarray, axis=0)
203            elif self.opr == 'median':
204                temp_dataarray = np.stack(temp_dataarray)
205                self.dataarray = np.median(temp_dataarray)
206            else:
207                raise Exception("No valid operation betwen data")
208        
209        else:
210            d, latitude, longitude = tif_to_array(self.dtf_names[0], self.dtf_path)
211            self.longitudearray = longitude
212            self.latitudearray = latitude
213            self.dataarray = d
